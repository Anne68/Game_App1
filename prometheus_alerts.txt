# 2.2 Créer prometheus/alerts.yml
cat > prometheus/alerts.yml << 'EOF'
groups:
  - name: model_alerts
    interval: 30s
    rules:
      # Alerte si le modèle n'est pas entraîné
      - alert: ModelNotTrained
        expr: model_games_count == 0
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "Model not trained"
          description: "The ML model has not been trained"
      
      # Alerte sur la latence élevée
      - alert: HighPredictionLatency
        expr: histogram_quantile(0.95, rate(prediction_latency_seconds_bucket[5m])) > 1
        for: 5m
        labels:
          severity: critical
        annotations:
          summary: "High prediction latency"
          description: "P95 latency is {{ $value }}s (threshold: 1s)"
      
      # Alerte sur le taux d'erreur
      - alert: HighErrorRate
        expr: |
          rate(model_predictions_total{status="error"}[5m]) / 
          rate(model_predictions_total[5m]) > 0.05
        for: 5m
        labels:
          severity: critical
        annotations:
          summary: "High error rate in predictions"
          description: "Error rate is {{ $value | humanizePercentage }}"

  - name: api_alerts
    interval: 30s
    rules:
      # API down
      - alert: APIDown
        expr: up{job="games-api"} == 0
        for: 1m
        labels:
          severity: critical
        annotations:
          summary: "API is down"
          description: "Games API has been down for {{ $value }} minutes"
EOF
